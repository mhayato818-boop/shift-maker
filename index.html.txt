async function sbFetch(path, method='GET', body=null) {
  if (!SUPABASE_URL || !SUPABASE_KEY) { toast('Supabase設定がありません', 'error'); return null; }
  const opts = {
    method,
    headers: {
      'apikey': SUPABASE_KEY,
      'Authorization': `Bearer ${SUPABASE_KEY}`,
      'Content-Type': 'application/json',
      'Prefer': method === 'POST' ? 'return=representation' : ''
    }
  };
  if (body) opts.body = JSON.stringify(body);
  try {
    const res = await fetch(`${SUPABASE_URL}/rest/v1/${path}`, opts);
    if (!res.ok) { const e = await res.json(); throw new Error(e.message || res.status); }
    if (res.status === 204) return [];
    return await res.json();
  } catch(e) {
    toast(`エラー: ${e.message}`, 'error');
    return null;
  }
}

function maybeShowGrid() {
  const staffId = document.getElementById('answer-staff').value;
  const periodId = document.getElementById('answer-period').value;
  if (!staffId || !periodId) { document.getElementById('answer-grid-wrap').style.display = 'none'; return; }
  const period = periods.find(p => p.id === periodId);
  if (!period) return;
  renderAvailGrid(period);
  document.getElementById('answer-grid-wrap').style.display = 'block';
}

let availSelections = {};

function renderAvailGrid(period) {
  availSelections = {};
  const { year, month } = period;
  const daysInMonth = new Date(year, month, 0).getDate();
  const grid = document.getElementById('answer-grid');
  let html = '';
  for (let d = 1; d <= daysInMonth; d++) {
    const date = new Date(year, month - 1, d);
    const dow = date.getDay();
    const isWeekend = dow === 0 || dow === 6;
    const dateStr = `${year}-${String(month).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    availSelections[dateStr] = 'ng';
    html += `
      <div class="avail-row">
        <div class="avail-date ${isWeekend ? 'weekend' : ''}">
          ${month}/${d}（${DAYS_JP[dow]}）
        </div>
        <div class="avail-btns">
          <div class="avail-btn" onclick="selectAvail('${dateStr}','ok',this)" data-date="${dateStr}">○</div>
          <div class="avail-btn" onclick="selectAvail('${dateStr}','maybe',this)" data-date="${dateStr}">△</div>
          <div class="avail-btn selected-ng" onclick="selectAvail('${dateStr}','ng',this)" data-date="${dateStr}">×</div>
        </div>
      </div>`;
  }
  grid.innerHTML = html;
}

function selectAvail(dateStr, status, btn) {
  availSelections[dateStr] = status;
  const row = btn.closest('.avail-btns');
  row.querySelectorAll('.avail-btn').forEach(b => b.className = 'avail-btn');
  btn.classList.add(`selected-${status}`);
}

async function submitAvailability() {
  const staffId = document.getElementById('answer-staff').value;
  const periodId = document.getElementById('answer-period').value;
  if (!staffId || !periodId) { toast('スタッフと期間を選択してください', 'error'); return; }

  // 既存削除

}

async function generateShift() {
  const periodId = document.getElementById('shift-period').value;
  if (!periodId) { toast('期間を選択してください', 'error'); return; }
  const period = periods.find(p => p.id === periodId);
  if (!period) return;

  document.getElementById('shift-content').innerHTML = '<div class="loading"><span class="spinner"></span>シフトを生成中...</div>';

  // データ取得
  const [availData, existingShifts] = await Promise.all([
    sbFetch(`availability?period_id=eq.${periodId}`),
    sbFetch(`shifts?period_id=eq.${periodId}`)
  ]);
  if (!availData) return;

  // 既存シフト削除
  if (existingShifts && existingShifts.length > 0) {
    await sbFetch(`shifts?period_id=eq.${periodId}`, 'DELETE');
  }

  const { year, month } = period;
  const daysInMonth = new Date(year, month, 0).getDate();
  const min_staff = period.min_staff;

  // 社員数チェック
  const employees = staffList.filter(s => s.role !== 'staff_part');
  const parts = staffList.filter(s => s.role === 'staff_part');
  const empCount = employees.length;

  // シフト生成
  const newShifts = [];

  for (let d = 1; d <= daysInMonth; d++) {
    const dateStr = `${year}-${String(month).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    const date = new Date(year, month - 1, d);
    const dow = date.getDay();
    const isWeekend = dow === 0 || dow === 6;

    const hnMin = min_staff.honten[dow];
    const ngMin = min_staff.nigoten[dow];

    // その日に○or△のスタッフを取得
    const availForDay = availData.filter(a => a.date === dateStr && (a.status === 'ok' || a.status === 'maybe'));
    const availStaffIds = availForDay.map(a => a.staff_id);

    // 店長は自店舗のみ
    const hontenManager = staffList.find(s => s.role === 'honten_manager' && availStaffIds.includes(s.id));
    const nigoteManager = staffList.find(s => s.role === 'nigoten_manager' && availStaffIds.includes(s.id));

    if (hontenManager) {
      newShifts.push({ period_id: periodId, staff_id: hontenManager.id, date: dateStr, store: 'honten', is_bridge: false });
    }
    if (nigoteManager) {
      newShifts.push({ period_id: periodId, staff_id: nigoteManager.id, date: dateStr, store: 'nigoten', is_bridge: false });
    }

    // 社員スタッフ
    const empStaffs = staffList.filter(s => s.role === 'staff_employee' && availStaffIds.includes(s.id));

    if (empCount >= 3) {
      // 社員3人以上 → スタッフ社員をブリッジ
      empStaffs.forEach(s => {
        newShifts.push({ period_id: periodId, staff_id: s.id, date: dateStr, store: s.store, is_bridge: true });
      });
    } else {
      // 社員2人以下 → 通常
      empStaffs.forEach(s => {
        newShifts.push({ period_id: periodId, staff_id: s.id, date: dateStr, store: s.store, is_bridge: false });
      });
      // アルバイト：社員2人+バイト1人の平日はバイト削除
      if (empCount === 2 && !isWeekend) {
        // バイトは入れない
      } else {
        // 土日またはその他 → バイトも追加
        parts.filter(s => availStaffIds.includes(s.id)).forEach(s => {
          newShifts.push({ period_id: periodId, staff_id: s.id, date: dateStr, store: s.store, is_bridge: false });
        });
      }
    }
  }

  // DB保存
  if (newShifts.length > 0) {
    const saved = await sbFetch('shifts', 'POST', newShifts);
    if (!saved) return;
  }

  toast('シフトを生成しました');
  await loadShift();
}

async function loadShift() {
  const periodId = document.getElementById('shift-period').value;
  if (!periodId) return;
  const period = periods.find(p => p.id === periodId);
  if (!period) return;

  const shifts = await sbFetch(`shifts?period_id=eq.${periodId}&order=date`);
  if (!shifts) return;
  currentShifts = shifts;

  renderShiftTable(period, shifts);
}

function renderShiftTable(period, shifts) {
  const { year, month } = period;
  const daysInMonth = new Date(year, month, 0).getDate();
  const min_staff = period.min_staff;

  // 日付ごとにシフトをグループ化
  const byDate = {};
  for (let d = 1; d <= daysInMonth; d++) {
    const dateStr = `${year}-${String(month).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    byDate[dateStr] = { honten: [], nigoten: [] };
  }
  shifts.forEach(s => {
    if (byDate[s.date]) byDate[s.date][s.store].push(s);
  });

  // 穴検出
  const shortages = [];
  Object.entries(byDate).forEach(([date, stores]) => {
    const d = new Date(date);
    const dow = d.getDay();
    const hnMin = min_staff.honten[dow];
    const ngMin = min_staff.nigoten[dow];
    if (stores.honten.length < hnMin) shortages.push({ date, store: '本店', need: hnMin, has: stores.honten.length });
    if (stores.nigoten.length < ngMin) shortages.push({ date, store: '2号店', need: ngMin, has: stores.nigoten.length });
  });

  // アラート
  const alertEl = document.getElementById('shift-alerts');
  if (shortages.length > 0) {
    // 連勤者を探して交換提案
    const suggestions = generateSwapSuggestions(shifts, shortages, byDate, min_staff, year, month, daysInMonth);
    alertEl.innerHTML = `
      <div class="alert alert-warn">
        ⚠️ 人手不足の日があります（${shortages.length}件）
      </div>
      ${suggestions.length > 0 ? `<div class="alert alert-info">
        💡 交換提案：<br>${suggestions.map(s => `・${s}`).join('<br>')}
      </div>` : ''}
    `;
  } else {
    alertEl.innerHTML = '<div class="alert alert-success">✓ すべての日程で最低人数を満たしています</div>';
  }

  // テーブル
  let html = `<div class="card shift-table-wrap"><table class="shift-table">
    <thead><tr>
      <th>日付</th>
      <th>曜日</th>
      <th style="color:var(--honten)">本店</th>
      <th style="color:var(--nigoten)">2号店</th>
      <th>状態</th>
    </tr></thead><tbody>`;

  Object.entries(byDate).forEach(([date, stores]) => {
    const d = new Date(date + 'T00:00:00');
    const dow = d.getDay();
    const isWeekend = dow === 0 || dow === 6;
    const hnMin = min_staff.honten[dow];
    const ngMin = min_staff.nigoten[dow];
    const hnShort = stores.honten.length < hnMin;
    const ngShort = stores.nigoten.length < ngMin;

    const dayNum = parseInt(date.split('-')[2]);

    const hnChips = stores.honten.map(s => {
      const staff = staffList.find(st => st.id === s.staff_id);
      return `<span class="day-chip honten">${staff ? staff.name : '?'}${s.is_bridge ? '🔀' : ''}</span>`;
    }).join('');
    const ngChips = stores.nigoten.map(s => {
      const staff = staffList.find(st => st.id === s.staff_id);
      return `<span class="day-chip nigoten">${staff ? staff.name : '?'}${s.is_bridge ? '🔀' : ''}</span>`;
    }).join('');

    const statusHtml = [
      hnShort ? `<span class="shortage">本店 ${stores.honten.length}/${hnMin}人</span>` : '',
      ngShort ? `<span class="shortage">2号店 ${stores.nigoten.length}/${ngMin}人</span>` : ''
    ].filter(Boolean).join(' ') || '<span style="color:var(--accent2);font-size:11px">✓</span>';

    html += `<tr ${(hnShort || ngShort) ? 'style="background:rgba(239,68,68,0.05)"' : ''}>
      <td style="font-family:'DM Mono',monospace;${isWeekend ? 'color:var(--warn)' : ''}">${date}</td>
      <td style="${isWeekend ? 'color:var(--warn)' : 'color:var(--text-muted)'}">${DAYS_JP[dow]}</td>
      <td><div class="day-staff">${hnChips || '<span style="color:var(--text-muted);font-size:11px">-</span>'}</div></td>
      <td><div class="day-staff">${ngChips || '<span style="color:var(--text-muted);font-size:11px">-</span>'}</div></td>
      <td>${statusHtml}</td>
    </tr>`;
  });

  html += '</tbody></table></div>';
  document.getElementById('shift-content').innerHTML = html;
}

function generateSwapSuggestions(shifts, shortages, byDate, min_staff, year, month, daysInMonth) {
  const suggestions = [];
  // 連勤チェック（3日以上連続）
  const staffShiftDates = {};
  shifts.forEach(s => {
    if (!staffShiftDates[s.staff_id]) staffShiftDates[s.staff_id] = [];
    staffShiftDates[s.staff_id].push(s.date);
  });

  shortages.slice(0, 3).forEach(shortage => {
    // 不足日の前後で連勤している人を探す
    Object.entries(staffShiftDates).forEach(([staffId, dates]) => {
      const sorted = dates.sort();
      // 3日以上連勤チェック
      for (let i = 0; i < sorted.length - 2; i++) {
        const d1 = new Date(sorted[i]);
        const d2 = new Date(sorted[i+1]);
        const d3 = new Date(sorted[i+2]);
        const diff1 = (d2 - d1) / 86400000;
        const diff2 = (d3 - d2) / 86400000;
        if (diff1 === 1 && diff2 === 1) {
          const staff = staffList.find(s => s.id === staffId);
          if (staff) {
            const storeLabel = shortage.store === '本店' ? '本店' : '2号店';
            suggestions.push(`${staff.name} の ${sorted[i+2]} を ${shortage.date}（${storeLabel}）と交換を検討`);
          }
        }
      }
    });
  });

  return [...new Set(suggestions)].slice(0, 5);
}

async function loadSwapData() {
  const periodId = document.getElementById('swap-period').value;
  if (!periodId) return;
  const shifts = await sbFetch(`shifts?period_id=eq.${periodId}&order=date`);
  if (!shifts) return;

  // スタッフ選択で日付を更新
  document.getElementById('swap-staff').addEventListener('change', () => updateSwapDates(shifts));
  updateSwapDates(shifts);
}

function updateSwapDates(shifts) {
  const staffId = document.getElementById('swap-staff').value;
  const staffShifts = shifts.filter(s => s.staff_id === staffId);
  const periodId = document.getElementById('swap-period').value;
  const period = periods.find(p => p.id === periodId);

  const fromOpts = staffShifts.map(s => {
    const d = new Date(s.date + 'T00:00:00');
    return `<option value="${s.id}">${s.date}（${STORE_LABELS[s.store]}）</option>`;
  }).join('');
  document.getElementById('swap-from-date').innerHTML = fromOpts || '<option value="">シフトなし</option>';

  // 変更先：その月の全日程
  if (period) {
    const { year, month } = period;
    const daysInMonth = new Date(year, month, 0).getDate();
    let toOpts = '';
    for (let d = 1; d <= daysInMonth; d++) {
      const dateStr = `${year}-${String(month).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
      toOpts += `<option value="${dateStr}">${dateStr}</option>`;
    }
    document.getElementById('swap-to-date').innerHTML = toOpts;
  }
}

async function applySwap() {
  const shiftId = document.getElementById('swap-from-date').value;
  const toDate = document.getElementById('swap-to-date').value;
  const toStore = document.getElementById('swap-to-store').value;
  const staffId = document.getElementById('swap-staff').value;
  if (!shiftId || !toDate || !staffId) { toast('全項目を入力してください', 'error'); return; }

  const staff = staffList.find(s => s.id === staffId);
  const fromShift = currentShifts.find(s => s.id === shiftId) || { date: '?' };

  swapData = { shiftId, toDate, toStore, staffId };

  document.getElementById('modal-swap-content').innerHTML = `
    <div class="alert alert-info">
      <strong>${staff ? staff.name : ''}</strong> のシフトを変更します<br>
      ${fromShift.date}（${STORE_LABELS[fromShift.store] || ''}）→ ${toDate}（${STORE_LABELS[toStore]}）
    </div>
  `;
  document.getElementById('modal-swap').classList.add('open');
}

async function confirmSwap() {
  const { shiftId, toDate, toStore } = swapData;
  const result = await sbFetch(`shifts?id=eq.${shiftId}`, 'PATCH', { date: toDate, store: toStore });
  if (result !== null) {
    closeModal('modal-swap');
    toast('シフトを変更しました');
    const periodId = document.getElementById('swap-period').value;
    if (periodId) {
      document.getElementById('shift-period').value = periodId;
      await loadShift();
    }
    // 履歴に追加
    const staff = staffList.find(s => s.id === swapData.staffId);
    swapHistory.unshift({ staff: staff?.name, toDate, toStore, at: new Date().toLocaleString('ja-JP') });
    renderSwapHistory();
  }
}

function renderSwapHistory() {
  const el = document.getElementById('swap-history');
  if (!swapHistory.length) { el.innerHTML = '<div class="empty">変更履歴はまだありません</div>'; return; }
  el.innerHTML = swapHistory.map(h => `
    <div style="padding:10px 0;border-bottom:1px solid var(--border);font-size:13px">
      <span style="color:var(--text)">${h.staff}</span> → 
      <span style="color:var(--accent)">${h.toDate}（${STORE_LABELS[h.toStore]}）</span>
      <span style="color:var(--text-muted);margin-left:8px;font-size:11px">${h.at}</span>
    </div>
  `).join('');
}
